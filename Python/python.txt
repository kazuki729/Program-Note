#venv仮想環境を作成

#プロジェクトフォルダ配下にvenv環境を構築するコマンド
> cd [プロジェクトフォルダ]
> python3 -m venv pythonapp-backend

#仮想環境有効化
> source pythonapp-frontend/bin/activate
#無効化
> deactivate


==バックエンド側の環境構築==
#仮想環境にchaliceをインストール
> pip install chalice

#python製のAWS SDKのBoto3インストール
> pip install SDKのBoto3

==フロントエンド側の環境構築==
> pip install transcrypt


==pipコマンド==
#pipでインストール済みのパッケージ一覧表示
> pip list
